
# FILE: ./.github/workflows/destroy.yml
name: Infrastructure Destruction

on:
  workflow_dispatch:

concurrency:
  group: terraform-deploy
  cancel-in-progress: false

jobs:
  terraform-destroy:
    runs-on: ubuntu-latest
    steps:
      - name: 코드 체크아웃
        uses: actions/checkout@v3

      - name: Terraform 설치
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: "1.7.0"
          terraform_wrapper: false

      - name: AWS 자격 증명 설정
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ap-northeast-2

      - name: Windows Server 내부 설정 찌꺼기 초기화 (SSM)
        run: |
          INSTANCE_ID=$(aws ssm describe-instance-information --query "InstanceInformationList[?PingStatus=='Online'].InstanceId" --output text)
          
          if [ -n "$INSTANCE_ID" ] && [ "$INSTANCE_ID" != "None" ]; then
            echo "온프레미스 서버($INSTANCE_ID)를 발견했습니다. 초기화 스크립트를 전송합니다."
            
            B64_SCRIPT=$(cat ./scripts/vpn_bgp_cleanup.ps1 | base64 -w 0)
            EXEC_CMD="[System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String('$B64_SCRIPT')) | Invoke-Expression"
            
            COMMAND_ID=$(aws ssm send-command \
              --document-name "AWS-RunPowerShellScript" \
              --targets "Key=instanceids,Values=$INSTANCE_ID" \
              --parameters "commands=[\"$EXEC_CMD\"]" \
              --query "Command.CommandId" \
              --output text)
              
            echo "명령 전송 완료. Command ID: $COMMAND_ID"
            aws ssm wait command-executed --command-id "$COMMAND_ID" --instance-id "$INSTANCE_ID"
          else
            echo "온라인 상태인 서버를 찾을 수 없어 내부 초기화를 건너뜁니다."
          fi

      - name: Terraform 초기화 및 삭제
        working-directory: ./terraform
        run: |
          terraform init
          terraform destroy -auto-approve

# FILE: ./.github/workflows/deploy.yml
name: Hybrid Cloud DR Provisioning

on:
  push:
    branches: [ "main" ]

concurrency:
  group: terraform-deploy
  cancel-in-progress: false

jobs:
  deploy-infra-and-configure:
    runs-on: ubuntu-latest
    steps:
      - name: 코드 체크아웃
        uses: actions/checkout@v3

      - name: Terraform 설치
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: "1.7.0"
          terraform_wrapper: false

      - name: AWS 자격 증명 설정
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ap-northeast-2

      - name: Terraform 초기화 및 배포
        working-directory: ./terraform
        run: |
          terraform init
          terraform apply -auto-approve

      - name: Terraform Output 추출 및 환경 변수 등록
        working-directory: ./terraform
        run: |
          # 1. 인스턴스가 Online이 될 때까지 최대 10번(200초) 재시도하는 로직
          echo "AWS SSM에서 Online 상태의 인스턴스를 찾는 중..."
          MAX_RETRIES=10
          RETRY_COUNT=0
          INSTANCE_ID="None"

          while [ $RETRY_COUNT -lt $MAX_RETRIES ]; do
            INSTANCE_ID=$(aws ssm describe-instance-information --query "InstanceInformationList[?PingStatus=='Online'].InstanceId | [0]" --output text | tr -d '\r\n')
            
            if [ "$INSTANCE_ID" != "None" ] && [ -n "$INSTANCE_ID" ] && [ "$INSTANCE_ID" != "null" ]; then
              echo "✅ 인스턴스 인식 성공: $INSTANCE_ID"
              break
            fi
            
            RETRY_COUNT=$((RETRY_COUNT+1))
            echo "⏳ 아직 인스턴스가 준비되지 않았습니다. (재시도 $RETRY_COUNT/$MAX_RETRIES)... 20초 후 다시 시도합니다."
            sleep 20
          done

          # 2. 최종 확인 후 환경 변수 등록
          if [ "$INSTANCE_ID" == "None" ] || [ -z "$INSTANCE_ID" ]; then
            echo "❌ 오류: 200초를 기다렸으나 Online 상태의 노드를 찾지 못했습니다."
            exit 1
          fi

          # 3. 안전한 환경 변수 기록 (줄바꿈 제거 tr -d '\r\n' 필수)
          {
            echo "INSTANCE_ID=$INSTANCE_ID"
            echo "TUNNEL1_IP=$(terraform output -raw vpn_tunnel1_ip | tr -d '\r\n')"
            echo "TUNNEL1_PSK=$(terraform output -raw vpn_tunnel1_psk | tr -d '\r\n')"
            echo "BGP_PEER1_IP=$(terraform output -raw vpn_bgp_peer1_ip | tr -d '\r\n')"
            echo "BGP_LOCAL1_IP=$(terraform output -raw vpn_bgp_local1_ip | tr -d '\r\n')"
            echo "TUNNEL2_IP=$(terraform output -raw vpn_tunnel2_ip | tr -d '\r\n')"
            echo "TUNNEL2_PSK=$(terraform output -raw vpn_tunnel2_psk | tr -d '\r\n')"
            echo "BGP_PEER2_IP=$(terraform output -raw vpn_bgp_peer2_ip | tr -d '\r\n')"
            echo "BGP_LOCAL2_IP=$(terraform output -raw vpn_bgp_local2_ip | tr -d '\r\n')"
          } >> $GITHUB_ENV
      - name: AWS SSM으로 Windows 설정 스크립트 전송 및 대기
        run: |
          cat <<EOF > final_script.ps1
          \$AwsVpcCidr = "10.1.0.0/16"
          \$OnpremVpcCidr = "192.168.0.0/24"
          \$Tunnel1Ip = "${{ env.TUNNEL1_IP }}"
          \$Tunnel1Psk = "${{ env.TUNNEL1_PSK }}"
          \$BgpPeer1Ip = "${{ env.BGP_PEER1_IP }}"
          \$BgpLocal1Ip = "${{ env.BGP_LOCAL1_IP }}"
          
          \$Tunnel2Ip = "${{ env.TUNNEL2_IP }}"
          \$Tunnel2Psk = "${{ env.TUNNEL2_PSK }}"
          \$BgpPeer2Ip = "${{ env.BGP_PEER2_IP }}"
          \$BgpLocal2Ip = "${{ env.BGP_LOCAL2_IP }}"
          EOF
          
          cat ./scripts/vpn_bgp_setup.ps1 >> final_script.ps1
          
          B64_SCRIPT=$(cat final_script.ps1 | base64 -w 0)
          
          EXEC_CMD="[System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String('$B64_SCRIPT')) | Invoke-Expression"
          
          COMMAND_ID=$(aws ssm send-command \
            --document-name "AWS-RunPowerShellScript" \
            --targets "Key=instanceids,Values=${{ env.INSTANCE_ID }}" \
            --parameters "commands=[\"$EXEC_CMD\"]" \
            --query "Command.CommandId" \
            --output text)
            
          echo "명령 전송 완료. Command ID: $COMMAND_ID"
          
          aws ssm wait command-executed --command-id "$COMMAND_ID" --instance-id "${{ env.INSTANCE_ID }}"

# ---------------------------------------------------------
# 2. VPC 및 기본 네트워크 구성

resource "aws_vpc" "aws_vpc" {
  cidr_block           = var.aws_vpc_cidr
  enable_dns_hostnames = true
  tags = { Name = "VPC-A-AWS-Cloud" }
}

resource "aws_subnet" "aws_subnet" {
  vpc_id            = aws_vpc.aws_vpc.id
  cidr_block        = cidrsubnet(var.aws_vpc_cidr, 8, 1)
  availability_zone = "${var.aws_region}a"
  tags = { Name = "Subnet-A-AWS" }
}

resource "aws_route_table" "aws_rt" {
  vpc_id = aws_vpc.aws_vpc.id
  tags = { Name = "RT-A-AWS" }
}

resource "aws_route_table_association" "aws_rt_assoc" {
  subnet_id      = aws_subnet.aws_subnet.id
  route_table_id = aws_route_table.aws_rt.id
}

# ---------------------------------------------------------
# 3. 하이브리드 커넥티비티 (TGW + CGW + VPN)
# ---------------------------------------------------------
resource "aws_ec2_transit_gateway" "tgw" {
  amazon_side_asn                 = var.aws_asn
  auto_accept_shared_attachments  = "enable"
  default_route_table_association = "enable"
  default_route_table_propagation = "enable" 
  tags = { Name = "TGW-Main" }
}

resource "aws_ec2_transit_gateway_vpc_attachment" "tgw_attach_vpc_a" {
  subnet_ids         = [aws_subnet.aws_subnet.id]
  transit_gateway_id = aws_ec2_transit_gateway.tgw.id
  vpc_id             = aws_vpc.aws_vpc.id
  tags = { Name = "TGW-Attach-VPC-A" }
}

resource "aws_route" "aws_to_onprem" {
  route_table_id         = aws_route_table.aws_rt.id
  destination_cidr_block = var.onprem_vpc_cidr
  transit_gateway_id     = aws_ec2_transit_gateway.tgw.id
  depends_on             = [aws_ec2_transit_gateway_vpc_attachment.tgw_attach_vpc_a]
}

resource "aws_customer_gateway" "cgw" {
  bgp_asn    = var.onprem_asn
  ip_address = data.aws_ssm_parameter.cgw_public_ip.value 
  type       = "ipsec.1"
  tags = { Name = "CGW-Home-Windows-Server" }
}

resource "aws_vpn_connection" "vpn" {
  customer_gateway_id = aws_customer_gateway.cgw.id
  transit_gateway_id  = aws_ec2_transit_gateway.tgw.id
  type                = aws_customer_gateway.cgw.type
  
  static_routes_only  = false 

  tunnel1_preshared_key = data.aws_ssm_parameter.tunnel1_psk.value
  tunnel1_inside_cidr   = "169.254.127.124/30"
  
  tunnel2_preshared_key = data.aws_ssm_parameter.tunnel2_psk.value
  tunnel2_inside_cidr   = "169.254.177.40/30"

  tags = { Name = "VPN-TGW-to-Home" }
}
